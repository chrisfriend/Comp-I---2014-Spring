\section{Response} % (fold)
\label{sec:response}
\begin{form}	
	\begin{enumerate}
		\item \TextField[multiline=true, width=\linewidth,height=2\baselineskip]{What is the main claim that the author(s) make(s)? What evidence can you show?\\}
		\item \TextField[multiline=true, width=\linewidth,height=4\baselineskip]{What are three pieces of evidence used by the author(s) to support this claim?\\}
		\item \TextField[multiline=true, width=\linewidth,height=4\baselineskip]{What do you think/feel/believe about the claims made by the author(s)? Are they valid? Important? Why (not)?\\}
		\item \TextField[multiline=true, width=\linewidth,height=5\baselineskip]{How does this text relate to another text used in this class (or in other courses/studies)? How are the authors of those two texts—not the texts themselves—interacting? (Consider whether they have compatible topics, direct references, common sources, shared theories/perspectives, etc.)\\}
		\item \TextField[multiline=true, width=\linewidth,height=4\baselineskip]{Considering the course objectives, what will you take from this reading as you move forward? Why is this significant toward your growth as a college writer?\\}
		\item \TextField[multiline=true, width=\linewidth,height=4\baselineskip]{What questions did this text prompt you to ask? List at least three that you think would make for interesting in-class conversation.\\}
	\end{enumerate}
\end{form}
